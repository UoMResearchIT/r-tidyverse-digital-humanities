gapminder <- read_csv("./data/gapminder-FiveYearData.csv", 
                      col_types = cols(
  country = col_character(),
  year = col_integer(),
  pop = col_double(),
  continent = col_character(),
  lifeExp = col_double(),
  gdpPercap = col_double()
)
            )

Lets investigate how life expectancy varies per continent:

gapminder %>% 
  filter(year == 1997) %>% 
  group_by(continent, year) %>% 
  summarise(medianLifeExp=median(lifeExp))

And lets plot a graph:

gapminder %>% filter(continent %in% c("Europe", "Americas")) %>% 
  ggplot(aes(x=year, y=gdpPercap, colour=country)) +
           geom_line() + facet_wrap( ~ continent) + guides(colour="none")
